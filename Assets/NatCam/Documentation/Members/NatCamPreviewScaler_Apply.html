NatCamPreviewScaler.Apply

public void Apply (Texture texture);

Description

Scale the UI Panel to conform to a texture.
 

NatCam.CapturePhoto((Texture2D photo) => {
    //Log
    Debug.Log("Received captured photo");
    //Enable the UI Panel
    photoRawImage.gameObject.SetActive(true);
    //Set the panel's texture to the photo we just captured
    photoRawImage.texture = photo;
    //Scale the UI panel
    photoRawImage.GetComponent<NatCamPreviewScaler>().OverrideApply(photo);
});

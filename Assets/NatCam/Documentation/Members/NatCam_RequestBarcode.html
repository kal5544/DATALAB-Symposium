NatCam.RequestBarcode

public static void RequestBarcode (BarcodeRequest[] requests);

Description

Request to be notified when NatCam detects a barcode. Note that when NatCam.Initialize is called, MetadataDetection must be on. It is off by default. 

void Start () {
    //Initialize NatCam with metadata detection
    NatCam.Initialize(MetadataDetection:Switch.On);
    //[Other initialization code...]
    //Start the camera preview
    NatCam.Play();
    //Create an ISBN request //You can request multiple barcode formats by using bitwise OR
    NatCam.RequestBarcode(new BarcodeRequest(OnDetectedISBN, BarcodeFormat.EAN_13 | BarcodeFormat.EAN_8, false));
}

void OnDetectedISBN (Barcode code) {
    //Log what we found
    Debug.Log("Found ISBN - " + code.ToString());
}

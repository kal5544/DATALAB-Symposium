NatCam.CapturePhoto

public static void CapturePhoto (PhotoCallback[] callbacks);

Description

Capture a photo and send it back to the provided callbacks. A PhotoCallback can be manually subscribed to the OnPhotoCapture event and will receive the captured photo.

Note that if this is called without any PhotoCallbacks, it will fail if there are no callbacks that have been manually subscribed to the OnPhotoCapture event.
 

void TakePicture () {
    //Register our photo capture delegate for NatCam's photo capture event
    NatCam.OnPhotoCapture += PhotoWasCaptured;
    //Capture a photo
    NatCam.CapturePhoto();
}

void PhotoWasCaptured (Texture2D photo) {
    //Set a material's main texture to be the captured photo
    material.mainTexture = photo;
    //Unregister our delegate from NatCam's photo capture event so that it doesn't get called if another photo is captured
    NatCam.OnPhotoCapture -= PhotoWasCaptured;
}
    

void TakePicture () {
    //Capture a photo with our delegate
    NatCam.CapturePhoto(PhotoWasCaptured); //We don't need to manually unregister this delegate
}

void PhotoWasCaptured (Texture2D photo) {
    //Set a material's main texture to be the captured photo
    material.mainTexture = photo;
}
    

void TakePicture () {
    //Capture a photo with our lambda delegate
    NatCam.CapturePhoto((Texture2D photo) => {
        //Set a material's main texture to be the captured photo
        material.mainTexture = photo;
        //We don't need to manually unregister this delegate
    });
}
